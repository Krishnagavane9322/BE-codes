//Design and implement Death First Search based on existing algorithms using OpenMP. Use a Tree or an undirected graph for BFS and DFS.
#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define MAX 100

int graph[MAX][MAX];   // Adjacency matrix
int visited[MAX];      // Visited array
int n;

// Parallel DFS function
void parallelDFS(int node) {

    // Critical to safely check/update visited
    #pragma omp critical
    {
        if (visited[node]) return;
        visited[node] = 1;
        printf("%d ", node);
    }

    // Explore neighbors in parallel
    #pragma omp parallel for
    for (int i = 0; i < n; i++) {
        if (graph[node][i] && !visited[i]) {
            parallelDFS(i);
        }
    }
}

int main() {
    int e;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    // Initialize
    for(int i = 0; i < n; i++) {
        visited[i] = 0;
        for(int j = 0; j < n; j++) {
            graph[i][j] = 0;
        }
    }

    printf("Enter number of edges: ");
    scanf("%d", &e);

    printf("Enter edges (u v):\n");
    for(int i = 0; i < e; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        graph[u][v] = 1;
        graph[v][u] = 1; // Undirected graph
    }

    int start;
    printf("Enter starting vertex: ");
    scanf("%d", &start);

    printf("\nParallel DFS Traversal:\n");

    // Start DFS in parallel region
    #pragma omp parallel
    {
        #pragma omp single
        {
            parallelDFS(start);
        }
    }

    return 0;
}
